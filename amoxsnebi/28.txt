#include <bits/stdc++.h>

using namespace std;

queue<int> q;

struct nod{
  int data;
  nod* next;
  
  nod(int data){
      this -> data = data;
      this -> next = NULL;
  }
  
};

nod* head;
nod* tail;

bool isEmpty(){
    return head == NULL;
}

void addToList(){
    nod* temp;
    if(isEmpty() && !q.empty()){
        temp = new nod(q.front());
        q.pop();
        head = tail = temp;
    }
    while(!q.empty()){
        temp = new nod(q.front());
        q.pop();
        tail -> next = temp;
        tail = temp;
    }
}

void print_data(){
    nod* temp = head;
    while(temp != NULL){
        cout << temp -> data << " ";
        temp = temp -> next;
    }
    
    cout << "\n";
}

int main()
{
    q.push(5);
    q.push(4);
    q.push(3);
    q.push(2);
    q.push(1);
    addToList();
    print_data();

    return 0;
}
