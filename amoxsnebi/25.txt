void putThird(int data){
    if (head == nullptr || head->next == nullptr) {
        return;
    }
    nod* temp = head->next; 
    nod* it = new nod(data);
    it->next = temp->next;
    temp->next = it;
}

// for double linked list
void insert_in(int n, int data) {  
    node* temp = head;  
    for (int i = 1; i < n - 1; ++i)    
        temp = temp -> next;  
        node* it = new node(data);  
        it->next = temp->next;  
        temp->next = it;  
        it->next->prev = it;  
        it->prev = temp;
}

// for single linked list
void insert_in(int n, int data) {  
    node* temp = head;  
    for (int i = 1; i < n - 1; ++i)    
        temp = temp -> next;  
        node* it = new node(data);  
        it->next = temp->next;  
        temp->next = it;  
}
