// for double linked list

void delete_in(int n) {  
    node* temp = head;  
    for (int i = 1; i < n; ++i)    
        temp = temp->next;  
        temp->prev->next = temp->next;  
        temp->next->prev = temp->prev;
}

void delete_in(int n) {  
    node* temp = head;
